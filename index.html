<script>

    let currentAnswer = 0;

    let userInput = "";

    let remainingQuestions = 10;

    const praiseMessages = ["すごい！", "天才！", "その調子！", "正解！", "やったね！", "ばっちり！"];


    // 問題を作成

    function nextQuestion() {

        if (remainingQuestions <= 0) {

            showFinish();

            return;

        }

        

        userInput = "";

        updateDisplay();

        

        const n1 = Math.floor(Math.random() * 10) + 1;

        const n2 = Math.floor(Math.random() * 10) + 1;

        const isPlus = Math.random() > 0.5;


        if (isPlus) {

            document.getElementById('num1').innerText = n1;

            document.getElementById('num2').innerText = n2;

            document.getElementById('operator').innerText = "+";

            currentAnswer = n1 + n2;

        } else {

            // 引き算は答えがマイナスにならないように

            const big = Math.max(n1, n2);

            const small = Math.min(n1, n2);

            document.getElementById('num1').innerText = big;

            document.getElementById('num2').innerText = small;

            document.getElementById('operator').innerText = "-";

            currentAnswer = big - small;

        }

    }


    // 数字ボタンが押されたとき

    function appendNumber(num) {

        if (userInput.length < 3) {

            userInput += num;

            updateDisplay();

        }

    }


    // クリアボタン

    function clearInput() {

        userInput = "";

        updateDisplay();

    }


    // 画面の更新

    function updateDisplay() {

        document.getElementById('user-input').innerText = userInput;

        document.getElementById('count').innerText = remainingQuestions;

    }


    // 答え合わせ

    function checkAnswer() {

        if (userInput === "") return;


        const box = document.getElementById('question-box');

        

        if (parseInt(userInput) === currentAnswer) {

            showPraise();

            remainingQuestions--;

            setTimeout(nextQuestion, 1000);

        } else {

            // 間違えたとき

            box.classList.add('shake');

            setTimeout(() => box.classList.remove('shake'), 300);

            userInput = "";

            updateDisplay();

        }

    }


    // 褒める演出

    function showPraise() {

        const feedback = document.getElementById('feedback');

        const msg = praiseMessages[Math.floor(Math.random() * praiseMessages.length)];

        feedback.innerText = msg;

        feedback.classList.add('show');

        

        setTimeout(() => {

            feedback.classList.remove('show');

        }, 800);

    }


    // 全問終了

    function showFinish() {

        document.getElementById('app').innerHTML = `

            <h1 style="font-size: 2.5rem;">✨ おめでとう！ ✨</h1>

            <p style="font-size: 1.2rem;">ぜんぶ クリアしたよ！</p>

            <button class="num-btn enter-btn" style="width: 100%; margin-top:20px;" onclick="location.reload()">もういちど やる</button>

        `;

    }


    // 最初の問題を表示

    nextQuestion();

</script>
